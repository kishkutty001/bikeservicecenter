import android.os.Bundle
import android.widget.Button
import android.widget.TextView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import androidx.recyclerview.widget.LinearLayoutManager
import androidx.recyclerview.widget.RecyclerView

class MechanicDashboardActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_mechanic_dashboard) // Change to your XML file name

        // Find views
        val tvWelcome = findViewById<TextView>(R.id.tvWelcome) // Needs to be added in XML
        val btnOnline = findViewById<Button>(R.id.btnOnline) // Needs to be added in XML
        val requestsRecyclerView = findViewById<RecyclerView>(R.id.requestsRecyclerView)

        // Example: set welcome text from shared preferences
        val prefs = getSharedPreferences("UserSession", MODE_PRIVATE)
        val name = prefs.getString("name", "Mechanic")
        tvWelcome?.text = "Welcome back, $name!"

        // Example: handle "Online" button click
        btnOnline?.setOnClickListener {
            Toast.makeText(this, "Status set to Online", Toast.LENGTH_SHORT).show()
        }

        // Setup RecyclerView with dummy data for now
        val dummyRequests = listOf(
            "Flat Tire - Customer A",
            "Engine Issue - Customer B"
        )

        requestsRecyclerView.layoutManager = LinearLayoutManager(this)
        requestsRecyclerView.adapter = RequestsAdapter(dummyRequests)
    }
}
